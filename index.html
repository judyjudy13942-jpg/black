let state = { baseQuota: 1500, carryOver: 0, totalExpense: 0, transactions: [] }; // 2. 從 Supabase 初始化讀取資料 async function initData() { try { // A. 抓取 profiles 資料（取得基本額度與上月結餘） const { data: profileData, error: pError } = await supabase .from('profiles') .select('*') .limit(1); if (pError) throw pError; if (profileData && profileData.length > 0) { myProfileId = profileData[0].id; // 記錄自己的 UUID state.baseQuota = parseFloat(profileData[0].monthly_quota); state.carryOver = parseFloat(profileData[0].carry_over); } // B. 抓取 transactions 當月記帳明細 const { data: transData, error: tError } = await supabase .from('transactions') .select('*') .order('created_at', { ascending: false }); // 新的在前面 if (tError) throw tError; if (transData) { state.transactions = transData.map(item => ({ title: item.title, amount: parseFloat(item.amount), date: new Date(item.created_at).toLocaleDateString('zh-TW') })); // 計算本月總消費（將所有記帳的金額加總） state.totalExpense = transData.reduce((sum, item) => sum + parseFloat(item.amount), 0); } // 重新渲染畫面 updateUI(); } catch (error) { console.error("讀取 Supabase 資料出錯：", error.message); alert("資料讀取失敗，請檢查 API 金鑰設定！"); } } // 3. 畫面渲染函數 (保持不變) function updateUI() { const currentQuota = state.baseQuota + state.carryOver - state.totalExpense; const quotaDisplay = document.getElementById('quotaDisplay'); quotaDisplay.innerText = `$${currentQuota.toLocaleString()}`; if (currentQuota < 0) { quotaDisplay.className = "quota-display quota-negative"; } else { quotaDisplay.className = "quota-display quota-positive"; } document.getElementById('baseQuota').innerText = `$${state.baseQuota}`; document.getElementById('carryOver').innerText = `$${state.carryOver}`; document.getElementById('totalExpense').innerText = `$${state.totalExpense}`; const listEl = document.getElementById('historyList'); if(state.transactions.length === 0) { listEl.innerHTML = `
  • 暫無紀錄
    • `; return; } listEl.innerHTML = state.transactions.map(item => `
  • ${item.title}
    ${item.date}
    ${item.amount > 0 ? `-${item.amount}` : `+${Math.abs(item.amount)}`}
    • `).join(''); } // 4. 表單送出事件（寫入資料到 Supabase） document.getElementById('transactionForm').addEventListener('submit', async (e) => { e.preventDefault(); const title = document.getElementById('title').value; const amount = parseFloat(document.getElementById('amount').value); try { // 👉 核心操作：將資料 Insert 到 Supabase const { data, error } = await supabase .from('transactions') .insert([ { user_id: myProfileId, title: title, amount: amount } ]); if (error) throw error; // 檢查是否爆額度並觸發 LINE 通知 const currentQuota = state.baseQuota + state.carryOver - (state.totalExpense + amount); if (currentQuota < 0) { sendLineNotification(title, amount, currentQuota); } // 清空輸入框並重新從資料庫抓取最新資料更新 UI document.getElementById('title').value = ''; document.getElementById('amount').value = ''; await initData(); } catch (error) { console.error("寫入資料失敗：", error.message); alert("記帳失敗：" + error.message); } }); // 5. LINE 通知發送邏輯 function sendLineNotification(title, amount, currentQuota) { const message = `\n⚠️ 記帳爆表通知！\n項目：${title}\n金額：$${amount}\n目前剩餘額度：$${currentQuota} (已成負數)`; console.log("🚨 觸發 LINE 通知：", message); // 這裡可以串接你的 LINE Notify 或是 LINE Bot API } // 網頁開啟時，觸發初始化讀取 initData();
